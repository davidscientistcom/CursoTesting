pytest en profundidad

fixtures · parametrize · marcas · conftest · plugins esenciales · pytest.ini

Por qué pytest y no unittest

Python incluye unittest en la librería estándar, pero el ecosistema industrial usa pytest. No es dogma — hay razones concretas:

pytestunittest
Sintaxis assert x == y — Python nativo, mensajes de error detallados automáticos self.assertEqual(x, y) — verboso, herencia de clase obligatoria
Fixtures Inyección por nombre, composables, con scope configurable setUp/tearDown por clase, no composables
Parametrización @pytest.mark.parametrize directamente sobre la función Requiere plugins externos o bucles manuales
Plugins Ecosistema de >1500 plugins (coverage, asyncio, django, xdist…) Ecosistema pequeño
Compatibilidad Ejecuta también tests de unittest — migración incremental posible Solo unittest
instalacion.sh — setup inicial de pytest en un proyecto 
# Instalación mínima
pip install pytest

# Stack recomendado para producción
pip install pytest pytest-cov pytest-asyncio pytest-xdist

# Verificar instalación

# pytest 8.x.x

# Ejecutar todos los tests

# Verbose — ver nombre de cada test

# Solo tests fallidos del último run
pytest --lf

# Parar al primer fallo
pytest -x

Fixtures — el corazón de pytest

Las fixtures son funciones que proveen contexto a los tests: objetos, conexiones, datos, configuración. La diferencia con setUp: las fixtures son modulares, composables y con lifecycle controlado por scope.

fixtures_basicas.py — fixture simple, yield para teardown 
import pytest
from dataclasses import dataclass, field


@dataclass
class BaseDeDatosEnMemoria:
    """Simula una BD para tests."""
    _datos: dict = field(default_factory=dict)

    def insertar(self, tabla: str, registro: dict) -> None:
        self._datos.setdefault(tabla, []).append(registro)

    def consultar(self, tabla: str) -> list:
        return self._datos.get(tabla, [])

    def limpiar(self) -> None:
        self._datos.clear()


# Fixture básica — se crea para cada test (scope="function" por defecto)
@pytest.fixture
def db():
    base = BaseDeDatosEnMemoria()
    yield base  # código antes del yield = setup, después = teardown
    base.limpiar()  # se ejecuta después de cada test, pase o falle


# Las fixtures se inyectan por nombre en la firma del test
def test_insertar_registro(db):
    db.insertar("usuarios", {"id": 1, "nombre": "Ana"})
    resultado = db.consultar("usuarios")
    assert len(resultado) == 1
    assert resultado[0]["nombre"] == "Ana"


def test_bd_empieza_limpia_en_cada_test(db):
    """Este test no ve los datos del test anterior — cada uno recibe su propia fixture."""
    resultado = db.consultar("usuarios")
    assert resultado == []


# Fixture con parámetros — puede recibir otras fixtures
@pytest.fixture
def db_con_usuarios(db):
    """Fixture de alto nivel que depende de otra fixture."""
    db.insertar("usuarios", {"id": 1, "nombre": "Ana"})
    db.insertar("usuarios", {"id": 2, "nombre": "Bob"})
    return db


def test_hay_dos_usuarios(db_con_usuarios):
    usuarios = db_con_usuarios.consultar("usuarios")
    assert len(usuarios) == 2
fixtures_scope.py — scope de fixture: cuándo se crea y destruye 
import pytest


# scope="function" (default): una instancia por test
@pytest.fixture
def conexion_funcion():
    print("  [OPEN function-scope]")
    yield {"conexion": "function"}
    print("  [CLOSE function-scope]")


# scope="module": una instancia compartida por todo el módulo/fichero
@pytest.fixture(scope="module")
def conexion_modulo():
    print("  [OPEN module-scope]")
    yield {"conexion": "module"}
    print("  [CLOSE module-scope]")


# scope="session": una instancia por toda la ejecución de pytest (todos los ficheros)
@pytest.fixture(scope="session")
def servidor_externo():
    """Útil para levantar un servidor de test una sola vez."""
    print("  [START server]")
    yield {"url": "http://localhost:8765"}
    print("  [STOP server]")


# ── Cuándo usar cada scope ─────────────────────────────────────────
# function  → datos mutables que no deben compartirse entre tests (la mayoría)
# module    → recursos caros pero que no se modifican (schemas, configuración)
# session   → recursos muy caros de levantar (servidores, conexiones externas reales)

def test_a(conexion_funcion):
    assert conexion_funcion["conexion"] == "function"

def test_b(conexion_funcion):
    # Recibe UNA NUEVA instancia — test_a no puede haberla contaminado
    assert conexion_funcion["conexion"] == "function"
WARN scope="session" con fixtures mutables es una fuente clásica de tests que solo fallan en ciertas órdenes. Si la fixture tiene estado que los tests modifican, usa scope="function". Si es costosa de crear y es de solo lectura (schema de BD, configuración SSL), entonces scope="session" es apropiado.

parametrize — eliminar duplicación en tests

@pytest.mark.parametrize ejecuta el mismo test con múltiples conjuntos de datos. Resultado: N casos de prueba con 1 función. Esencial para testear casos borde sin duplicar código.

parametrize_basico.py — misma lógica, múltiples entradas 
import pytest


def es_palindromo(texto: str) -> bool:
    t = texto.lower().replace(" ", "")
    return t == t[::-1]


# SIN parametrize — repetición, difícil de mantener
def test_radar_es_palindromo():
    assert es_palindromo("radar") is True

def test_hola_no_es_palindromo():
    assert es_palindromo("hola") is False


# CON parametrize — limpio, extensible, cada caso tiene su propio nombre
@pytest.mark.parametrize("texto,esperado", [
    ("radar",              True),
    ("Amo la paloma",      True),   # case-insensitive + espacios
    ("hola",              False),
    ("",                   True),   # vacío es palíndromo trivial
    ("A",                  True),   # un solo carácter
    ("ab",                False),
])
def test_es_palindromo(texto, esperado):
    assert es_palindromo(texto) == esperado


# ── parametrize con IDs personalizados ───────────────────────────
@pytest.mark.parametrize("numero,primo", [
    pytest.param(2,  True,  id="dos-es-primo"),
    pytest.param(17, True,  id="diecisiete-es-primo"),
    pytest.param(4,  False, id="cuatro-no-es-primo"),
    pytest.param(1,  False, id="uno-no-es-primo"),
    pytest.param(0,  False, id="cero-no-es-primo"),
])
def test_es_primo(numero, primo):
    def _es_primo(n):
        if n < 2:
            return False
        return all(n % i != 0 for i in range(2, int(n**0.5) + 1))

    assert _es_primo(numero) == primo


# ── parametrize combinatorio (producto cartesiano) ──────────────
@pytest.mark.parametrize("base", [2, 8, 16])
@pytest.mark.parametrize("exponente", [0, 1, 2])
def test_potencia_nunca_negativa(base, exponente):
    assert base ** exponente >= 0

Marcas (marks) — organizar y controlar la ejecución

Las marcas permiten categorizar tests, saltarlos condicionalmente y filtrar qué tests ejecutar en cada contexto (PR rápido vs CI completo).

marcas_y_skip.py — skip, skipif, xfail, marcas personalizadas 
import pytest
import sys


# ── skip — siempre se salta ──────────────────────────────────────
@pytest.mark.skip(reason="WIP: lógica de descuento aún no implementada")
def test_descuento_premium_pendiente():
    assert False  # Este fallo no cuenta


# ── skipif — salta condicionalmente ─────────────────────────────
@pytest.mark.skipif(sys.platform == "win32", reason="Paths POSIX no disponibles en Windows")
def test_lectura_fichero_posix():
    assert "/tmp/test".startswith("/")


# ── xfail — fallo esperado (el test documenta un bug conocido) ───
@pytest.mark.xfail(reason="Bug #247: parseo de fechas ISO con zona horaria")
def test_parseo_fecha_con_timezone():
    from datetime import datetime
    # Este fallo es conocido — el test lo documenta pero no bloquea el build
    resultado = datetime.fromisoformat("2024-01-15T10:30:00+02:00")
    assert resultado.hour == 10  # fallará en versiones antiguas de Python


# ── Marcas personalizadas — para filtrar tests en CLI ────────────
@pytest.mark.integration
def test_conexion_base_de_datos():
    """Requiere BD real. Se ejecuta solo en CI con --integration."""
    pass


@pytest.mark.slow
def test_proceso_batch_completo():
    """Tarda 10s. Excluido del ciclo rápido de desarrollo."""
    pass


@pytest.mark.smoke
def test_api_health_check():
    """Test rápido de humo. Siempre se ejecuta."""
    pass
pytest.ini — configuración del proyecto (alternativa: pyproject.toml) 
[pytest]
# Dónde buscar tests
testpaths = tests

# Patrones de ficheros de test
python_files = test_*.py *_test.py
python_classes = Test*
python_functions = test_*

# Marcas registradas (evita warnings de marcas no declaradas)
markers =
    integration: tests que requieren infraestructura real
    slow: tests que tardan más de 5 segundos
    smoke: tests de humo para verificación rápida
    unit: tests unitarios puros

# Opciones por defecto cuando se ejecuta pytest sin argumentos
addopts =
    -v
    --tb=short
    --strict-markers

# ── Uso en CLI ──────────────────────────────────────────────────
# Solo tests rápidos (excluir slow e integration):
#   pytest -m "not slow and not integration"
#
# Solo smoke tests:
#   pytest -m smoke
#
# Solo unit tests:
#   pytest -m unit
#
# Con coverage:
#   pytest --cov=src --cov-report=term-missing

conftest.py — fixtures compartidas entre ficheros

conftest.py es descubierto automáticamente por pytest. Las fixtures que defines ahí están disponibles en todos los tests del mismo directorio y subdirectorios, sin necesidad de importarlas.

conftest.py — fichero central de fixtures compartidas 
# tests/conftest.py
#
# Estructura del proyecto:
#   src/
#     auth.py
#     orders.py
#   tests/
#     conftest.py          ← fixtures compartidas por todos los tests
#     test_auth.py
#     test_orders.py
#     integration/
#       conftest.py        ← fixtures solo para tests de integración
#       test_db.py

import pytest
from unittest.mock import MagicMock
from src.auth import ServicioAuth
from src.orders import OrderService


# ── Fixtures de dominio compartidas ─────────────────────────────

@pytest.fixture
def mock_user_repo():
    repo = MagicMock()
    repo.verificar_credenciales.return_value = True
    return repo


@pytest.fixture
def servicio_auth(mock_user_repo):
    return ServicioAuth(repo=mock_user_repo)


@pytest.fixture
def usuario_autenticado(servicio_auth):
    """Token de un usuario autenticado — útil en múltiples tests."""
    return servicio_auth.login("ana@ejemplo.com", "password123")


# ── Fixture de datos de prueba ───────────────────────────────────

@pytest.fixture
def pedido_de_ejemplo():
    return {
        "id": "ORD-001",
        "total": 250.0,
        "email": "ana@ejemplo.com",
        "items": [
            {"sku": "P001", "qty": 2, "precio": 100.0},
            {"sku": "P002", "qty": 1, "precio": 50.0},
        ]
    }


# ── autouse — fixture que se aplica a todos los tests del scope ──

@pytest.fixture(autouse=True)
def reset_configuracion_global():
    """Asegura que la configuración esté limpia antes de cada test."""
    from src import config
    config.reset()
    yield
    config.reset()

async_test_ejemplo.py — testear código asyncio con pytest-asyncio 
# pytest.ini: asyncio_mode = auto

import asyncio
import pytest


async def obtener_precio_externo(producto_id: str) -> float:
    """Simula llamada async a API de precios."""
    await asyncio.sleep(0)  # yield de control
    precios = {"P001": 99.99, "P002": 49.99}
    return precios.get(producto_id, 0.0)


async def calcular_total(items: list[tuple[str, int]]) -> float:
    """Calcula el total de una lista de (producto_id, cantidad)."""
    total = 0.0
    for producto_id, cantidad in items:
        precio = await obtener_precio_externo(producto_id)
        total += precio * cantidad
    return total


# Con asyncio_mode = auto basta con async def test_...
async def test_calcular_total_dos_productos():
    items = [("P001", 2), ("P002", 1)]
    total = await calcular_total(items)
    assert total == pytest.approx(249.97)  # 99.99*2 + 49.99*1


async def test_producto_desconocido_suma_cero():
    items = [("DESCONOCIDO", 5)]
    total = await calcular_total(items)
    assert total == 0.0
mocker_fixture.py — pytest-mock: forma idiomática de mockear en pytest 
import smtplib
from email.message import EmailMessage


def enviar_email_bienvenida(email: str) -> bool:
    """Función de producción que usa smtplib."""
    try:
        with smtplib.SMTP("localhost", 1025) as server:
            msg = EmailMessage()
            msg["From"] = "noreply@empresa.com"
            msg["To"] = email
            msg["Subject"] = "Bienvenido"
            msg.set_content("Gracias por registrarte.")
            server.send_message(msg)
        return True
    except Exception:
        return False


# Con pytest-mock: la fixture 'mocker' se inyecta automáticamente
def test_enviar_email_llama_a_smtp(mocker):
    mock_smtp = mocker.patch("smtplib.SMTP")
    mock_smtp_instance = mock_smtp.return_value.__enter__.return_value

    resultado = enviar_email_bienvenida("ana@ejemplo.com")

    assert resultado is True
    mock_smtp_instance.send_message.assert_called_once()


def test_enviar_email_devuelve_false_si_smtp_falla(mocker):
    mocker.patch("smtplib.SMTP", side_effect=ConnectionRefusedError)

    resultado = enviar_email_bienvenida("ana@ejemplo.com")

    assert resultado is False
SUCCESS Configuración mínima para un proyecto real: pytest + pytest-cov + pytest-mock. Con eso cubre el 90% de los casos. Añade pytest-asyncio si tienes código async, pytest-xdist si la suite tarda más de 30 segundos, y freezegun si manejas fechas en la lógica de negocio.
